// THE BRAIN FUNCTION
// ============================================================================
export const getDinoResponse = (dinoName, input) => {
  const lowerInput = input.toLowerCase();

  // 1. GLOBAL CHECKS (Does everyone know this word?)
  for (const [key, responses] of Object.entries(GLOBAL_TRIGGERS)) {
    if (lowerInput.includes(key)) {
       return responses[Math.floor(Math.random() * responses.length)];
    }
  }

  // 2. FIND DINO & ARCHETYPE
  // We match partial names so "Rexy (Age 22)" matches "Rexy"
  let dinoKey = Object.keys(DINOS).find(k => dinoName.includes(k));
  // If we can't find the dino, default to Rexy logic
  if (!dinoKey) dinoKey = "Rexy"; 
  
  const dino = DINOS[dinoKey];
  const archetype = ARCHETYPES[dino.type];

  // 3. SPECIFIC DINO LORE (Priority 1)
  for (const [trigger, responses] of Object.entries(dino.specifics)) {
    if (lowerInput.includes(trigger)) {
      return responses[Math.floor(Math.random() * responses.length)];
    }
  }

  // 4. ARCHETYPE TRIGGERS (Priority 2)
  for (const [trigger, responses] of Object.entries(archetype.triggers)) {
    if (lowerInput.includes(trigger)) {
      return responses[Math.floor(Math.random() * responses.length)];
    }
  }

  // 5. CHAOS FALLBACK (Priority 3)
  // If nothing matched, say something random based on personality
  return archetype.defaults[Math.floor(Math.random() * archetype.defaults.length)];
};
